# NAME

Nephia::Plugin::CSRFDefender - CSRF Defender Plugin for Nephia

# SYNOPSIS

    package MyApp.pm;
    use strict;
    use warnings;
    use Nephia plugins => qw[
        'CSRFDefender'
    ];

# DESCRIPTION

Nephia::Plugin::CSRFDefender denies CSRF request.
